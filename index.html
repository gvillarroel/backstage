PlantUML · Sequence diagram

Model Approval Workflow

@startuml
!theme plain
actor "Team Lead" as TL
participant "Backstage" as BS
participant "Eval Engine" as EE
participant "Risk Council" as RC
database "Model Registry" as MR

TL -> BS : Submit model request
BS -> EE : Trigger evaluation
EE -> EE : Run benchmarks
EE -> BS : Return eval report

BS -> RC : Submit for review
RC -> RC : Assess risk & compliance
RC -> BS : Approval decision

alt Approved
  BS -> MR : Register model
  BS -> TL : Notify approved
else Rejected
  BS -> TL : Notify with feedback
end
@enduml
